#include <Keypad.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Servo.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

Servo doorServo;

// HC-05
SoftwareSerial BT(10, 11); // RX, TX

// Servo Pin
const int servoPin = 9;

// Keypad
const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] =
{
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {2,3,4,5};
byte colPins[COLS] = {6,7,8,12};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// Variables
String studentID = "";
String enteredOTP = "";
String generatedOTP = "";

bool waitingForOTP = false;
unsigned long startTime = 0;
const unsigned long otpTimeout = 30000;

void setup()
{
  Serial.begin(9600);
  BT.begin(9600);

  lcd.init();
  lcd.backlight();

  doorServo.attach(servoPin);
  doorServo.write(0);

  randomSeed(analogRead(A0));

  lcd.setCursor(0,0);
  lcd.print("Smart Door Lock");
  delay(2000);

  resetSystem();
}

void loop()
{
  if(waitingForOTP)
  {
    unsigned long remaining =
      otpTimeout - (millis() - startTime);

    if(millis() - startTime >= otpTimeout)
    {
      lcd.clear();
      lcd.print("OTP Expired");

      Serial.println("ACCESS FAILED : OTP Timeout");

      delay(2000);
      resetSystem();
      return;
    }

    lcd.setCursor(0,1);
    lcd.print("T:");
    lcd.print(remaining/1000);
    lcd.print("s   ");
  }

  char key = keypad.getKey();

  if(key)
  {
    if(!waitingForOTP)
    {
      handleIDInput(key);
    }
    else
    {
      handleOTPInput(key);
    }
  }
}

void handleIDInput(char key)
{
  if(key>='0' && key<='9')
  {
    if(studentID.length()<4)
    {
      studentID += key;

      lcd.setCursor(studentID.length()-1,1);
      lcd.print("*");
    }
  }

  else if(key=='*')
  {
    studentID="";

    lcd.clear();
    lcd.print("Enter 4-digit");
    lcd.setCursor(0,1);
  }

  else if(key=='#')
  {
    if(studentID.length()==4)
    {
      generateOTP();

      waitingForOTP=true;
      startTime=millis();

      lcd.clear();
      lcd.print("Enter OTP");
      lcd.setCursor(0,1);

      Serial.print("Student ID : ");
      Serial.println(studentID);

      Serial.print("Generated OTP : ");
      Serial.println(generatedOTP);

      BT.print("Your OTP: ");
      BT.println(generatedOTP);

      enteredOTP="";
    }
  }
}

void handleOTPInput(char key)
{
  if(key>='0' && key<='9')
  {
    if(enteredOTP.length()<6)
    {
      enteredOTP+=key;

      lcd.setCursor(enteredOTP.length()-1,0);
      lcd.print("*");
    }
  }

  else if(key=='*')
  {
    enteredOTP="";

    lcd.clear();
    lcd.print("Enter OTP");
